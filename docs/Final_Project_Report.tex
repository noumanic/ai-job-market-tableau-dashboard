\chapter*{Abstract}
This report documents an end-to-end data visualization project on the global Artificial Intelligence (AI) and Machine Learning (ML) labour market for 2022--2025. Working from a real, crowdsourced dataset of \textbf{147,348} salary submissions across \textbf{87 countries} and \textbf{15 standardized role categories}, we built a Python preprocessing pipeline that enriches the raw data with twelve engineered analytical columns, and an interactive Tableau dashboard that brings ten coordinated worksheets and five headline KPIs onto a single \(1200 \times 1500\) canvas. The dashboard answers five guiding questions: which roles pay the most, how compensation grows across career levels, how remote work has shifted, where the highest-paying hiring activity is concentrated, and how salaries are distributed across pay bands. This report walks through the dataset, the preprocessing pipeline, twenty exploratory figures, every Tableau worksheet, the headline insights, the value Tableau added to the analysis, and full source code in the appendix.

\chapter{Introduction}

\section{Project Context}
The AI/ML job market is one of the most rapidly evolving sectors of the global economy. Generative AI and large language models have re-shaped which roles command top compensation, where hiring is concentrated, and how teams work. This project, ``AI Job Market Intelligence,'' converts a real crowdsourced salary dataset into an interactive, story-driven Tableau dashboard so readers can navigate this evolution at a glance.

\section{Guiding Questions}
The dashboard and report are organised around five questions derived from the project proposal:
\begin{enumerate}
    \item Which AI/ML/Data Science roles pay the most?
    \item How does salary grow from Entry Level to Executive?
    \item How has remote work changed year over year from 2022 to 2025?
    \item Which countries lead global AI hiring, and how do salaries differ by region?
    \item What does the salary distribution look like across pay bands and quartiles?
\end{enumerate}

\section{Objectives}
\begin{itemize}
    \item \textbf{Data Integrity:} Clean and normalize crowdsourced salary data into an analysis-ready table.
    \item \textbf{Feature Engineering:} Add twelve derived columns (region, salary band, role-relative deviations, etc.) that unlock richer Tableau encodings.
    \item \textbf{Exploratory Analysis:} Document twenty matplotlib/seaborn figures that profile every facet of the market.
    \item \textbf{Dashboarding:} Deliver an interactive Tableau master dashboard with ten coordinated worksheets, five KPI tiles, four insight callouts, and a global cross-filter on country.
\end{itemize}

\chapter{About the Dataset}
\label{ch:dataset}

\section{What was the dataset about?}
\label{sec:what_dataset}
The dataset is the \textit{Global Salaries in AI, ML, Data Science and Big Data} survey published under \textbf{CC0 1.0 Public Domain} by \texttt{aijobs.net}, the same first-party organisation that operates the live job board. It is a real, crowdsourced compensation survey rather than a synthetic or scraped third-party file. Every row is a single, anonymous self-reported salary submission. Together, the records describe what AI/ML/Data Science professionals are paid worldwide between 2022 and 2025.

\textbf{Each record carries eleven raw fields:} the work year, the experience level (Entry, Mid, Senior, Executive), the employment type, the free-text job title, the local-currency salary, the currency code, the salary converted to USD, the employee's country of residence, the remote-work ratio (0/50/100), the company's country, and the company size (Small/Medium/Large). After preprocessing we standardise the 406 raw job titles into 15 categories and add twelve engineered analytical columns, bringing the final table to 27 columns and 147,348 cleaned rows.

\subsection*{Why this dataset is real, not synthetic}
\begin{itemize}[leftmargin=*]
    \item Published on Kaggle under the official \texttt{aijobs} organisation account — a first-party release, not a re-upload.
    \item Real-world distribution markers: the United States accounts for 90\% of submissions (matching public hiring statistics), salary distributions are right-skewed (skewness 0.68), and full-time employment is 99.4\% of records.
    \item 406 distinct raw job-title strings reflect genuine free-text variation rather than a clean synthetic enumeration.
    \item Updated weekly by the same platform that runs the live aijobs.net job board, so the data is grounded in active hiring activity.
\end{itemize}

\chapter{Preprocessing Pipeline}

\section{Methodology}
The pipeline is designed for scalability and reproducibility. Five Python scripts transform the raw 151,445 records into a refined 147,348-row table with 27 analytical columns. No synthetic data is introduced at any stage --- every row in the final output is a real salary submission.

\section{Pipeline Stages}
\begin{enumerate}
    \item \texttt{01\_explore.py} --- initial profiling: shape, nulls, distributions, skewness.
    \item \texttt{02\_clean.py} --- filtering and outlier removal.
    \item \texttt{03\_title\_grouping.py} --- mapping 406 raw job titles into 15 standardized categories.
    \item \texttt{04\_summary\_stats.py} --- computing every KPI metric used in the dashboard.
    \item \texttt{05\_feature\_engineering.py} --- engineering twelve new analytical columns.
\end{enumerate}

\section{Cleaning Stages and Row Accounting}
\begin{table}[H]
    \centering
    \begin{tabular}{lrrr}
        \toprule
        \textbf{Step} & \textbf{Rows Before} & \textbf{Rows After} & \textbf{Removed} \\
        \midrule
        Keep Full-Time only & 151,445 & 150,541 & 904 \\
        Keep 2022--2025 only & 150,541 & 150,264 & 277 \\
        Outlier removal (1st--99th percentile) & 150,264 & 147,348 & 2,916 \\
        \bottomrule
    \end{tabular}
    \caption{Row accounting through the cleaning pipeline.}
\end{table}

\section{Engineered Columns}
Twelve analytical columns were added without introducing synthetic data:
\begin{itemize}[leftmargin=*]
    \item \textbf{Categorical bins} — \texttt{salary\_band} (six pay tiers), \texttt{salary\_quartile} (Q1--Q4).
    \item \textbf{Geographic enrichment} — \texttt{region} (six regions), \texttt{is\_us} (US vs Rest of World).
    \item \textbf{Role family} — \texttt{is\_ai\_core} (AI/ML Core vs broader Data \& Tech).
    \item \textbf{Sort key} — \texttt{exp\_order} so the experience axis renders Entry $\to$ Mid $\to$ Senior $\to$ Executive.
    \item \textbf{Year-relative} — \texttt{year\_avg\_salary}, \texttt{salary\_vs\_year\_avg}, \texttt{salary\_vs\_year\_avg\_pct}.
    \item \textbf{Role-relative} — \texttt{role\_avg\_salary}, \texttt{salary\_vs\_role\_avg}, \texttt{salary\_vs\_role\_avg\_pct}.
\end{itemize}

\chapter{Key Insights --- What We Found}
\label{ch:insights}

\section{Insight 1: ML Engineering is the highest-paid AI/ML track}
\begin{insightbox}
ML Engineer / MLOps leads at \$194,303 average --- 81\% above the lowest-paid track (Data Analyst at \$107,465). Software Engineer (\$180K) and Research Scientist (\$175K) follow.
\end{insightbox}

\section{Insight 2: Compensation grows nearly $2\times$ from Entry to Executive}
\begin{insightbox}
Entry Level \$102,743 $\to$ Mid \$140,382 $\to$ Senior \$170,351 $\to$ Executive \$195,935. The EX/EN ratio is exactly $1.91\times$, and the inter-quartile spread widens with seniority (\$60K at Entry vs \$99K at Executive).
\end{insightbox}

\section{Insight 3: Remote work has collapsed since 2022}
\begin{insightbox}
Remote dropped from \textbf{53.5\% (2022)} to \textbf{20.3\% (2025)}. On-site nearly doubled from 43.5\% to 79.6\%. Hybrid effectively vanished from 3\% to under 0.05\%. The dataset documents the end of the remote-first era for AI roles.
\end{insightbox}

\section{Insight 4: The US dominates volume, but the premium is everywhere on the map}
\begin{insightbox}
The United States supplies \textbf{90.4\% of submissions} (133,143 of 147,348) at a \$161,381 average. Rest of World averages \$108,024. North America commands a 94\% premium over Latin America, the lowest-paying region.
\end{insightbox}

\section{Insight 5: Salaries are heavily concentrated in the \$100K--\$200K band}
\begin{insightbox}
55\% of all records sit in the combined \$100K--\$200K window (31\% in \$100K--\$150K plus 24\% in \$150K--\$200K). The top tier (\$260K+) holds 7.3\% --- a healthy upper tail. Sub-\$60K accounts for under 4\%.
\end{insightbox}

\section{Insight 6: AI/ML Core roles command a 13\% premium}
\begin{insightbox}
AI/ML Core (engineers, researchers, MLOps) averages \$171,009 vs \$150,685 for the broader Data \& Tech family. Direct AI/ML exposure is rewarded with a $\sim$\$20K annual premium even after holding role family constant.
\end{insightbox}

\section{Insight 7: Company size matters --- but only past ``Small''}
\begin{insightbox}
Small companies pay \$103K avg; Medium \$155K; Large \$165K. The largest jump is between Small and Medium ($\sim$50\% lift); Medium-to-Large is only \$10K. Diversifying into Medium-sized firms gives roughly the same pay as Large with a far wider job-supply.
\end{insightbox}

\section{Insight 8: The market is maturing --- volume keeps climbing, pay has plateaued}
\begin{insightbox}
Submission volume grew 48$\times$ between 2022 (1,587) and 2025 (76,424). Yet the median salary trajectory flattens after 2024 (\$149K $\to$ \$146K). The market is broadening more than it is inflating.
\end{insightbox}

\chapter{How Tableau Helped in Our Analysis}
\label{ch:tableau_helped}

While the EDA in Python answered \textit{what} the data shows, Tableau answered \textit{so what} --- it took the static findings and made them explorable, comparable, and cross-filterable on a single canvas. This chapter enumerates the concrete ways Tableau elevated the analysis.

\section{1. From static figures to a single interactive canvas}
The twenty Python figures each told one story. Tableau merged the strongest ten into a unified $1200\times1500$ dashboard with cross-filtering, so a reader can investigate a specific cohort (e.g.\ ``Senior, US, 2024'') without producing a new figure. The KPI tiles update live as filters are applied.

\section{2. Click-to-filter actions across every worksheet}
A single \texttt{Filter} action on the Global Map propagates the selected country into every other worksheet. Clicking the United States re-scopes the role bar, the experience bars, the work-mode donut, and all four comparative panels. This kind of synchronized cross-analysis is impractical to replicate in matplotlib.

\section{3. Custom calculated fields and parameters}
\begin{itemize}[leftmargin=*]
    \item \textbf{Avg Salary USD:} \texttt{AVG([Salary In Usd])} --- standardised the metric across every worksheet so KPI tiles and chart axes stay consistent.
    \item \textbf{Median Avg Salary USD:} \texttt{MEDIAN([Salary In Usd])} --- used to overlay the median line on the trend chart.
    \item \textbf{Year filter:} a multi-value dropdown wired to every sheet via \texttt{Apply to Worksheets All Using This Data Source}.
    \item \textbf{Custom shelf sorts} on \texttt{salary\_band} and \texttt{title\_group} so the bars render in salary order rather than alphabetical.
\end{itemize}

\section{4. Geographic intelligence out of the box}
By tagging \texttt{company\_location} with the \textit{Country/Region} geographic role, Tableau auto-resolved the 87 ISO-2 codes into a world choropleth without writing a single line of GIS code. Mapbox tiles and the colour-by-AVG encoding let the reader spot the US dominance instantly.

\section{5. Tooltips that beat static legends}
Every panel has a custom tooltip showing role count, average, median, and (where relevant) deviation from the year or role average. Hovering replaces the need for legends and side tables.

\section{6. KPI strip drives at-a-glance reading}
Five colour-coded KPI tiles --- median, mean, total records, remote \%, EX/EN multiple --- sit directly under the title bar. A reader gets the ``shape'' of the market in two seconds before drilling into any chart.

\section{7. Visual encoding richness}
\begin{itemize}[leftmargin=*]
    \item \textbf{Sequential colour scales} on the choropleth and the Salary By Role bar make magnitude visible at a glance.
    \item \textbf{Categorical palettes} on Work Mode and Salary Band Distribution use colour to encode dimension membership.
    \item \textbf{Dual-axis} on Salary Trend lets a single chart compare two measures (median line + volume bars) on a shared time axis.
\end{itemize}

\section{8. Reproducibility through a packaged \texttt{.twb} file}
The entire dashboard is committed as a single \texttt{Dashboard.twb} XML file in the repository. Anyone with Tableau Desktop can open it, repoint the data connection, and reproduce every chart, calculated field, filter and action exactly. There is no hidden state.

\section{Summary}
Tableau converted our static findings into a tool. Without it the report would be a slide deck of charts; with it, the report has a companion artefact that the reader can manipulate. That shift --- from \textit{communicate} to \textit{explore} --- is the principal value Tableau brought to this project.

This project converted a real, crowdsourced dataset of 147,348 AI/ML/Data Science salary submissions into an interactive Tableau master dashboard plus a reproducible Python preprocessing pipeline. import pandas as pd

def group_title(title: str) -> str:
    t = title.lower()
    if any(x in t for x in ["data scientist", "data science", "applied scientist"]):
        return "Data Scientist"
    elif any(x in t for x in ["data engineer", "data engineering", "etl"]):
        return "Data Engineer"
    elif any(x in t for x in ["machine learning engineer", "ml engineer", "mlops"]):
        return "ML Engineer / MLOps"
    elif any(x in t for x in ["data analyst", "bi analyst", "reporting analyst"]):
        return "Data Analyst"
    elif any(x in t for x in ["ai engineer", "deep learning", "nlp engineer"]):
        return "AI / ML Researcher"
    else:
        return "Other"

df = pd.read_csv("../data/processed/salaries_clean.csv")
df["title_group"] = df["job_title"].apply(group_title)
df["work_mode"]  = df["remote_ratio"].map({0: "On-site", 50: "Hybrid", 100: "Remote"})
df.to_csv("../data/processed/salaries_clean_final.csv", index=False)
\end{lstlisting}

\section{05\_feature\_engineering.py}
\begin{lstlisting}[language=Python]
import pandas as pd
import numpy as np

df = pd.read_csv("../data/processed/salaries_clean_final.csv")

# 1. SALARY BAND
bins   = [0, 60_000, 100_000, 150_000, 200_000, 260_000, 385_001]
labels = ["<$60K", "$60K-$100K", "$100K-$150K", "$150K-$200K", "$200K-$260K", "$260K+"]
df["salary_band"] = pd.cut(df["salary_in_usd"], bins=bins, labels=labels)

# 2. SALARY QUARTILE
df["salary_quartile"] = pd.qcut(
    df["salary_in_usd"], q=4,
    labels=["Q1 - Bottom 25%", "Q2 - 25-50%", "Q3 - 50-75%", "Q4 - Top 25%"]
)

# 3. REGION
region_map = {"US": "North America", "CA": "North America", "GB": "Europe", ...}
df["region"] = df["company_location"].map(region_map).fillna("Other")

# 4. SALARY vs ROLE AVERAGE
role_avg = df.groupby("title_group")["salary_in_usd"].mean()
df["role_avg_salary"]        = df["title_group"].map(role_avg).round(0)
df["salary_vs_role_avg"]     = (df["salary_in_usd"] - df["role_avg_salary"]).round(0)
df["salary_vs_role_avg_pct"] = (df["salary_vs_role_avg"] / df["role_avg_salary"] * 100).round(1)

df.to_csv("../data/processed/salaries_enhanced.csv", index=False)
\end{lstlisting}
